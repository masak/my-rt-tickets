use v6;

loop (my $n = 2; $n <= 354294; $n++)
{
my @digits = $n.comb(/\d/);
my $sum = [+] (@digits >>**>> 5);
say "$n, $sum" if ($n % 1234 == 0);
say "$n = sum of its digits to the 5th power!!!" if ($sum == $n);
}
