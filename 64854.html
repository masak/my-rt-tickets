+=item chdir STRING
+
+Changes the current working directory.
+
+ chdir '/new/dir';
+
+On Success the value of the new directory is put in $*CWD.
+
+=cut
+
+.sub 'chdir'
+ .param string newdir
+ .local pmc os
+ .local int success
+
+ # Error with no arguments.
+ $I0 = length newdir
+ if $I0 goto ok
+ 'die'("Cannot call chdir without an argument")
+ ok:
+
+ os = new 'OS'
+ success = 0
+
+ # Get CWD
+ $S0 = os.'cwd'()
+
+ # skip if already there
+ if $S0 == newdir goto noneed
+
+ # Try to set CWD
+ os.'chdir'(newdir)
+
+ # See if CWD has changed
+ $S1 = os.'cwd'()
+ if $S0 == $S1 goto finish
+
+ $P0 = new 'Str'
+ $P0 = $S1
+ set_hll_global '$CWD', $P0
+ noneed:
+ inc success
+ finish:
+
+ .return (success)
+.end
+
=back

=cut
